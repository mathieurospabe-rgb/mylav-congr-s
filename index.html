// ============================================== // Google Apps Script — MYLAV Contacts Congrès // VERSION CORRIGÉE — Compatible ancien et nouveau HTML // ============================================== // Colonnes du Sheet : // A: Clinique | B: Contact | C: Adresse mail | D: Ville // E: Code postal | F: Commentaires | G: Commercial référent // H: Statut du contact | I: Action réalisée | J: Date du dernier contact // K: Prochaine action | L: CA 2025 var SHEET_NAME = 'Contacts congrès 2026'; // Adapter si le nom de l'onglet est différent // ---- GET : lire les contacts ---- function doGet(e) { var action = (e && e.parameter && e.parameter.action) ? e.parameter.action : ''; if (action === 'getContacts') { return getContacts(); } return ContentService.createTextOutput(JSON.stringify({ status: 'ok' })) .setMimeType(ContentService.MimeType.JSON); } function getContacts() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.getActiveSheet(); } var data = sheet.getDataRange().getValues(); if (data.length <= 1) { return ContentService.createTextOutput(JSON.stringify([])) .setMimeType(ContentService.MimeType.JSON); } // Colonnes : A=Clinique, B=Contact, C=Email, D=Ville, E=CP, F=Commentaires, G=Commercial, H=Statut var contacts = []; for (var i = 1; i < data.length; i++) { var row = data[i]; // Ignorer les lignes vides (pas de contact) if (!row[1]) continue; // Ignorer les lignes de séparation/titre (colonne A remplie mais B vide) if (row[0] && !row[1]) continue; contacts.push({ row: i + 1, clinique: row[0] || '', contact: row[1] || '', email: row[2] || '', ville: row[3] || '', codepostal: row[4] || '', commentaires: row[5] || '', commercial: row[6] || '', statut: row[7] || '' }); } return ContentService.createTextOutput(JSON.stringify(contacts)) .setMimeType(ContentService.MimeType.JSON); } // ---- POST : créer ou modifier un contact ---- function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.getActiveSheet(); } var data; try { data = JSON.parse(e.postData.contents); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: 'Invalid JSON' })) .setMimeType(ContentService.MimeType.JSON); } // Gérer le champ contact : soit "contact" (nouveau HTML), soit "prenom"+"nom" (ancien HTML) var contactName = ''; if (data.contact) { contactName = data.contact; } else if (data.prenom || data.nom) { contactName = ((data.prenom || '') + ' ' + (data.nom || '')).trim(); } // Gérer le champ commentaires : soit "commentaires" (nouveau HTML), soit "note" (ancien HTML) var commentaires = data.commentaires || data.note || ''; // Construire la ligne de 12 colonnes (A à L) — identique à l'ancien script var values = [ data.clinique || '', // A: Clinique contactName, // B: Contact data.email || '', // C: Adresse mail data.ville || '', // D: Ville data.codepostal || '', // E: Code postal commentaires, // F: Commentaires data.commercial || '', // G: Commercial référent data.statut || '', // H: Statut du contact '', // I: Action réalisée (vide) new Date().toLocaleDateString('fr-FR'), // J: Date du dernier contact '', // K: Prochaine action (vide) '' // L: CA 2025 (vide) ]; if (data.action === 'update' && data.row) { // Modifier une ligne existante — écrire uniquement les 8 premières colonnes // pour ne pas écraser les colonnes I à L remplies par les commerciaux var rowNum = parseInt(data.row); if (rowNum >= 2 && rowNum <= sheet.getLastRow()) { var updateValues = [ data.clinique || '', contactName, data.email || '', data.ville || '', data.codepostal || '', commentaires, data.commercial || '', data.statut || '' ]; var range = sheet.getRange(rowNum, 1, 1, 8); // Colonnes A à H seulement range.setValues([updateValues]); } } else { // Nouvelle ligne — écrire les 12 colonnes sheet.appendRow(values); } return ContentService.createTextOutput(JSON.stringify({ status: 'success' })) .setMimeType(ContentService.MimeType.JSON); }
